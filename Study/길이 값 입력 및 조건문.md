## 길이 값 입력 및 조건문

  <학습목표>

* 사용자가 직접 값을 입력하고 결과를 계산하는 방식을 경험한다.
* 사용자가 입력한 문자열을 변환하거나 일부를 비교하는 방법을 경험한다.
* 비교문 결과에 따라 다른 동작을 하도록 개선한다.
* 옵셔널 데이터 예외처리 방식에 익숙해진다.

```swift

import Foundation

// 함수1 : meter 단위제거 : trimmingCharacters사용
func eraseMeterUnit (_ input: String) -> Double{
    var num: Double = 0.0
    num += Double(input.trimmingCharacters(in: ["m"]))!
    return num
}

// centi 단위제거 : prefix upto와 index사용
func eraseCentiUnit (_ input: String) -> Double{
    var num: Double = 0.0
    num += Double(input.prefix(upTo: input.index(of: "c")!))!
    return num
}

// centi를 meter로 변환
func convertCentiToM (input: Double) -> String {
    var result: String = ""
    result += String(input/100) + "m"
    return result
}

// meter를 centi로 변환
func convertMeterToCenti (input: Double) -> String {
    var result: String = ""
    result += String(input*100) + "cm"
    return result
}

// 결과를 출력
func printResult (_ result: String) {
    print(result)
}

print ("변환하고 싶은 단위를 입력해주세요.")
let input = readLine() ?? "에러"
var result = ""
if input.hasSuffix("cm") {
   result = convertCentiToM(input: eraseCentiUnit(input))
} else {
   result = convertMeterToCenti(input: eraseMeterUnit(input))
}
printResult(result)
```
